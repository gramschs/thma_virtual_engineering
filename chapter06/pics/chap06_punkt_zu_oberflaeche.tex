  \foreach \x/\y in {0/0.3, 2.5/0, 5.5/0.4}{

  \filldraw[fill=my_orange, draw=my_lightgray, line width=1.2pt]

    at (2.75,-1.6) {\ding{55}};
  \node[font=\small\bfseries, text=my_darkgray, anchor=north, align=center]
    at (2.75,-1.9) {Abstand zum\\nächsten Vertex};
